Build your neural network model in object pascal, easily like never before. Start training the model in a simple pipeline:

Prepare the data

{ Load and prepare the data. }
Dataset := ReadCSV('iris.csv');
Enc     := TOneHotEncoder.Create;    

X := GetColumnRange(Dataset, 0, 4);
X := StandardScaler(X);

y := GetColumn(Dataset, 4);
y := Enc.Encode(Squeeze(y));

{ Wrap tensor in a TVariable }
XVar := X.ToVariable(False);
yVar := y.ToVariable(False);

Build the model

MyModel := TModel.Create([
    TDenseLayer.Create(4, 20, atReLU), // Fully-connected layer with ReLU
    TDenseLayer.Create(20, 3, atNone), // Fully-connected layer
    TSoftMaxLayer.Create(1)            // Softmax layer
]);

{ Noe provides implementation of several useful
  optimization algorithms. }
opt := TAdamOptimizer.Create;
opt.LearningRate := 0.01;

Start the training

for i := 0 to 100 do
begin
    yPred := MyModel.Eval(X);               // Make a prediction
    Loss  := CrossEntropyLoss(yPred, y);    // Compute the loss

    opt.UpdateParams(Loss, MyModel.Params); // Update the model parameters
end;
